#INPUT

Create empty Dictionary called 'config'
Open "config.txt" file in reading mode

FOR EACH 'line' in the file:
    'key', 'value' <- Separate 'line' by the character "="
    'config'[key] <- Convert 'value' to decimal number#float
END FOR

Close "config.txt" file

#PROCESS

#Forzar to integers the variables that represent counts or dimensions
width <- Convert to Integer (config["width"] )
high <- Convert to Integer (config["high"] )
max_iter <- Convert to Integer (config["max_iter"] )

Open "class.csv" file in write mode (as 'output')
Write the line "row + column + iterations" in 'output'

FOR 'row' FROM 0 TO (high - 1):
    FOR 'column' FROM 0 TO (width - 1):
        
        # Map the coordinates (x,y) of the screen to coordinates of the complex plane (real, imaginary)
        real <- config["real_min"] + (column /width) * (config["real_max"] - config["real_min"])
        imag <- config["imag_min"] + (row /height) * (config["imag_max"] - config["imag_min"])
        
        c <- Create complex number with parts (real, imag)
        z <- Create complex number (0, 0)
        iterations = 0
        
        #Bucle fractal evaluation (Exhaust condition)
        WHILE (Absolute Value of 'z' <= 2) Y ('iterations' < max_iter) DO:
            z <- (z * z) + c
            iterations <- iterations + 1
        END WHILE
        
        
        #Salida partial loop
        
        Write the values ('row'+ 'column' + 'iterations') separated by commas in 'output'
        
    END FOR
END FOR

Close 'output' file


#OUTPUT

Display "DONE"

END
