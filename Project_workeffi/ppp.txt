# INPUT
task_list <- [] 
keep_adding <- true

WHILE keep_adding == true DO:
    Get task_name, task_time, unit (hours/minutes), important_level from user
    
    # Estandarización inmediata
    IF unit == "hours" THEN: task_time <- task_time * 60
    
    # Crear objeto de tarea
    new_task <- {name: task_name, time: task_time, level: important_level, date: urgency_date, status: "active"}
    Add new_task to task_list
    
    Ask user: "Add another task?"
    ...
END WHILE

# MÓDULO DE EDICIÓN (Sugerencia de feedback)
Ask user: "¿Deseas modificar alguna tarea? (yes/no)"
IF answer == "yes" THEN:
    Display task_list
    Get task_id_to_edit
    Update selected task_data (time, level, or date)
END IF

# PROCESS: Lógica de ajuste inteligente
Get total_time_available, unit_available
IF unit_available == "hours" THEN: total_time_available <- total_time_available * 60

level_map <- {"high": 3, "medium": 2, "low": 1}
# Incorporar urgencia en el valor de prioridad (a mayor urgencia, mayor valor)
FOR EACH task IN task_list:
    task.priority_score <- level_map[task.level] + (if task.date is soon then 1 else 0)

time_total_necessary <- sum of all task.time

IF time_total_necessary > total_time_available THEN:
    missing_time <- time_total_necessary - total_time_available
    
    # Lógica de posponer (Sacrificio inteligente)
    Sort task_list by priority_score (ascending) and time (descending)
    
    time_recovered <- 0
    FOR EACH task IN task_list:
        IF time_recovered < missing_time AND task.level != "high" THEN:
            task.status <- "postponed"
            time_recovered <- time_recovered + task.time
            # Sugerencia de ajuste (Feedback: dar tiempo de ajuste)
            Display "Sugiero posponer: ", task.name, ". Ganarás ", task.time, " minutos."
    
    IF time_recovered < missing_time THEN:
        Display "ALERTA: Incluso posponiendo tareas bajas y medias, falta tiempo."
END IF

# OUTPUT
Display "RESUMEN DE PLANIFICACIÓN"
FOR EACH task IN task_list:
    IF task.status == "active" THEN:
        Display task.name, " (Programada)"
    ELSE:
        Display task.name, " (Pospuesta para mañana)"
